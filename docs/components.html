Components

All available UI components in WindUI

Core Components

Window

Create the main UI window container

WindUI:CreateWindow({...})

Tab

Create tabs within a window

Window:Tab({...})

Section

Organize elements into sections

Tab:Section({...})

Interactive Elements

Button

Clickable button with icon and callback

Tab:Button({...})

Toggle

On/off toggle switch

Tab:Toggle({...})

Dropdown

Select from a list of options (supports dynamic updates)

Tab:Dropdown({...})

Input

Text input field

Tab:Input({...})

Slider

Numeric slider for values

Tab:Slider({...})

Keybind

Capture keyboard input

Tab:Keybind({...})

Colorpicker

Color selection picker

Tab:Colorpicker({...})

Display Elements

Paragraph

Text paragraph with optional buttons

Tab:Paragraph({...})

Code

Display code snippets

Tab:Code({...})

Image

Display images with aspect ratio control

Tab:Image({...})

Divider

Visual separator line

Tab:Divider()

Space

Add spacing between elements

Tab:Space({...})

Group

Group elements horizontally

Tab:Group({...})

Example Usage

-- Create window
local Window = WindUI:CreateWindow({
    Title = "My Script",
    Author = "by Developer",
    Folder = "myscript"
})

-- Create tab
local Tab = Window:Tab({
    Title = "Main",
    Icon = "home"
})

-- Add button
Tab:Button({
    Title = "Click Me",
    Icon = "mouse-pointer-click",
    Callback = function()
        print("Clicked!")
    end
})

-- Add toggle
local toggle = Tab:Toggle({
    Title = "Enable Feature",
    Value = false,
    Callback = function(value)
        print("Toggle:", value)
    end
})

-- Add dropdown (with dynamic updates)
local dropdown = Tab:Dropdown({
    Title = "Select Option",
    Values = {
        { Title = "Option 1" },
        { Title = "Option 2" }
    },
    Callback = function(option)
        print("Selected:", option.Title)
    end
})

-- Update dropdown dynamically
dropdown:SetValues({
    { Title = "New Option A" },
    { Title = "New Option B" }
})
