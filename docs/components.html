Detailed Examples

Complete Window Setup

-- Load WindUI
local WindUI = loadstring(game:HttpGet('https://raw.githubusercontent.com/ScriptsForDays/CUSTOMUI/refs/heads/main/dist/main.lua'))()

-- Create window with custom open button
local Window = WindUI:CreateWindow({
    Title = "My Script",
    Author = "by Developer",
    Folder = "myscript",
    Icon = "settings",
    NewElements = true,
    OpenButton = {
        Title = "Open UI",
        Enabled = true,
        Draggable = true
    }
})

-- Add window tag
Window:Tag({
    Title = "v1.0",
    Icon = "github",
    Color = Color3.fromHex("#6b31ff")
})

-- Create tabs
local MainTab = Window:Tab({
    Title = "Main",
    Icon = "home"
})

local SettingsTab = Window:Tab({
    Title = "Settings",
    Icon = "settings"
})

Interactive Elements

-- Button with highlight
local button = MainTab:Button({
    Title = "Click Me",
    Desc = "This button does something",
    Icon = "mouse-pointer-click",
    Color = Color3.fromHex("#30ff6a"),
    Callback = function()
        button:Highlight()
        print("Button clicked!")
    end
})

-- Toggle with state management
local toggle = MainTab:Toggle({
    Title = "Enable Feature",
    Desc = "Turn this feature on/off",
    Value = false,
    Type = "Toggle",
    Callback = function(value)
        print("Feature enabled:", value)
    end
})

-- Slider with range
local slider = MainTab:Slider({
    Title = "Volume",
    Desc = "Adjust volume level",
    Value = {
        Min = 0,
        Max = 100,
        Default = 50
    },
    Step = 1,
    Callback = function(value)
        print("Volume:", value)
    end
})

-- Input field
local input = MainTab:Input({
    Title = "Player Name",
    Placeholder = "Enter player name...",
    Type = "Input",
    Callback = function(value)
        print("Input:", value)
    end
})

-- Dropdown with dynamic updates (Custom Enhancement)
local dropdown = MainTab:Dropdown({
    Title = "Select Option",
    Values = {
        { Title = "Option 1", Icon = "file" },
        { Title = "Option 2", Icon = "folder" }
    },
    SearchBarEnabled = true,
    Callback = function(option)
        print("Selected:", option.Title)
    end
})

-- Update dropdown dynamically later
task.wait(5)
dropdown:SetValues({
    { Title = "New Option A", Icon = "star" },
    { Title = "New Option B", Icon = "heart" },
    { Title = "New Option C", Icon = "zap" }
})

Advanced Components

-- Colorpicker with explicit default color
local colorpicker = MainTab:Colorpicker({
    Title = "Pick Color",
    Default = Color3.fromHex("#30ff6a"),
    Callback = function(color)
        print("Color:", color)
    end
})

-- Colorpicker that displays current UI color (ThemeProperty)
local accentColorpicker = MainTab:Colorpicker({
    Title = "Accent Color",
    Desc = "Customize the accent color",
    ThemeProperty = "Accent", -- Automatically gets current Accent color
    Callback = function(color)
        WindUI:SetCustomAccent(color)
        -- UI updates immediately!
    end
})

-- Keybind
local keybind = MainTab:Keybind({
    Title = "Toggle Key",
    Value = Enum.KeyCode.F,
    Callback = function(key)
        print("Key pressed:", key)
    end
})

-- Section with collapsible content
local section = MainTab:Section({
    Title = "Advanced Settings",
    Icon = "settings",
    Opened = false,
    Box = true
})

-- Paragraph with buttons
MainTab:Paragraph({
    Title = "Information",
    Desc = "This is some important information",
    Buttons = {
        { Title = "Learn More", Icon = "info", Callback = function() end },
        { Title = "Close", Icon = "x", Callback = function() end }
    }
})

-- Image display
MainTab:Image({
    Image = "rbxassetid://123456789",
    AspectRatio = "16:9",
    Radius = 9
})

-- Group elements horizontally
MainTab:Group({
    Elements = {
        { Type = "Button", Title = "Button 1" },
        { Type = "Button", Title = "Button 2" },
        { Type = "Button", Title = "Button 3" }
    }
})
